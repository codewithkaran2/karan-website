    void main(){
        int arr[5], i,brr[5];
        for (i = 0; i < 5; i++)
        {
            printf("Enter Value  %d\n", i+1);
            scanf("%d", &arr[i]);
        }
    
        for (i = 0; i < 5; i++)
        {
            brr[4-i] = arr[i];
        }
        printf("Print Copied Array Values\n");

        for (i = 0; i < 5; i++)
        {
              printf("Value at  %d is %d\n", i+1, brr[i]);
        }
        return 0;
    }
