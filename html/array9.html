    void sort(int arr[], int size){
        int i , j;
        for (i = 0; i < size; i++){
            for (j = i+1; j < size; j++){
                if (arr[i] > arr[j]){
                    int temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                }
            }
        }
    }
    void main(){
        int i, arr[5], k;
        for (i = 0; i < 5; i++){
            printf("Enter Value of position %d\n", i+1);
            scanf("%d", &arr[i]);
        }
    
        printf("Enter K-th Value\n");
        scanf("%d", &k);
        
        if (k >= 1 && k <=5){
            sort(arr, 5);
            printf("K-th Smallest Value of given list is %d", arr[k-1]);
        }
        else
            printf("We have only 5 element list");
        return 0;
    }
